  function ordenar() {
    var versosOriginales = document.querySelectorAll('span');
    var versosOrdenados  = {};

    for(var i=0; i<versosOriginales.length; i++) {
      var clase = versosOriginales[i].classList[0];
      var n     = parseInt(clase.split('-')[1]);

      versosOrdenados[n] = versosOriginales[i];
    }

    var p = document.querySelector('p')
    p.innerHTML = '';

    for(var j in versosOrdenados)
      p.appendChild(versosOrdenados[j]);
  }

Combination Constructor/Prototype Pattern

    
  function Persona() {
    this.nombre = 'Paquito';
  }

  Persona.prototype = {
    constructor: Persona,
    is: function(nombre) {
      return this.nombre === nombre;
    }
  }


    
  var persona1 = new Persona;
  var persona2 = new Persona();
  
  persona1.nombre = 'Macarena';
  
  persona1.is('Paquito'); // false
  persona2.is('Paquito'); // true

Patrón básicos de herencia


Combination Inheritance // Pseudoclassical inheritance

  function SuperTipo(nombre) {
    this.nombre = nombre;
  }

  SuperTipo.prototype.diNombre = function() { alert(this.nombre); };

  function SubTipo(nombre, edad) {
    SuperTipo.call(this, nombre);
    this.edad = edad;
  }

  SubTipo.prototype = new SuperTipo();
  SubTipo.prototype.diEdad = function() { alert(this.edad); };
  
  var obj = new SubTipo('Paco', 32);
